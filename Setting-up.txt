# Setting up:
# 1 Install conda:
# - WSL / Ubuntu:
wget https://repo.continuum.io/archive/Anaconda3-2025.06-1-Linux-x86_64.sh
bash Anaconda3-2025.06-1-Linux-x86_64.sh
echo 'export PATH="$HOME/anaconda3/bin:$PATH"'  >> ~/.bashrc
# re-open cmd

# - MacOS:
https://repo.anaconda.com/archive/ choose type of CPU
bash Anaconda3-2025.06-VERSION.sh

# re-open cmd

# 2 Create conda virtual environment
conda create -n CSB2025 -c colomoto -c potassco -c conda-forge ginsim-python bns-python boolsim-python pymaboss notebook seaborn numpy pandas matplotlib

conda init
# re-open cmd

# 3 activate conda virtual environment 
conda activate CSB2025
pip install ipylab
